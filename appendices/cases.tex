\section{Kasus und Artikel}

\subsection{Bestimmte Artikel}

\begin{center}
\begin{tabular}{lllll}
\toprule
Kasus & Maskulin & Feminin & Neutrum & Plural \\
\midrule
Nominativ & der & die & das & die \\
Akkusativ & den & die & das & die \\
Dativ & dem & der & dem & den + n \\
Genitiv & des + s/es & der & des + s/es & der \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Unbestimmte Artikel}

\begin{center}
\begin{tabular}{lllll}
\toprule
Kasus & Maskulin & Feminin & Neutrum & Plural \\
\midrule
Nominativ & ein & eine & ein & -- \\
Akkusativ & einen & eine & ein & -- \\
Dativ & einem & einer & einem & -- \\
Genitiv & eines + s/es & einer & eines + s/es & -- \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Merksatz}

\begin{itemize}
    \item Akkusativ ändert nur maskulin: der $\rightarrow$ den, ein $\rightarrow$ einen.
    \item Dativ Plural bekommt meistens ein zusätzliches -n: mit den Kindern.
    \item Genitiv maskulin und neutrum bekommen meistens -s oder -es: des Tages, des Kindes.
\end{itemize}
